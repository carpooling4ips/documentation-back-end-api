/**
 * The main application file.
 *
 * @module
 */

'use strict'

/*
 * ==================================================
 * | Imports                                        |
 * ==================================================
 */
import path from 'path'
import { fileURLToPath } from 'url'
import express from 'express'
import rfs from 'rotating-file-stream'
import morgan from 'morgan'
import helmet from 'helmet'
import cors from 'cors'
import HttpStatus from 'http-status-codes'
import i18n from 'i18n'
import I18nMongooseError from 'mongoose-i18n-error'
import debug from 'debug'
import passport from 'passport'
import passportJwt from 'passport-jwt'
import luxon from 'luxon'
import mongoose from 'mongoose'
import cron from 'node-cron'

import NotificationManager from './utils/notificationManager.js'
import RideManager from './utils/rideManager.js'
import ApiError from './utils/apiError.js'

// Routers
import { DebugRouter, AuthRouter, UserRouter } from './routes/index.js'

/*
 * ==================================================
 * | Constants                                      |
 * ==================================================
 */
const { ReasonPhrases, StatusCodes } = HttpStatus
const { Strategy: StrategyJwt, ExtractJwt } = passportJwt
const { Settings: LuxonSettings } = luxon

// Set __filename and __dirname constants
const __filename = fileURLToPath(import.meta.url)
const __dirname = path.dirname(__filename)

// Debug package
const debugApp = debug('carpooling4ips')
const debugApi = debugApp.extend('api')

/*
 * ==================================================
 * | Initialize Express Framework                   |
 * ==================================================
 */

/**
 * Initialize cron jobs.
 */
export function initializeCronJobs () {
  // Set up ride manager check expirations cron job
  RideManager.checkExpirations()
  cron.schedule('0 * * * *', () => {
    RideManager.checkExpirations()
  })
}

/**
 * Initialize cron jobs.
 */
export function initializeServices () {
  // Initialize notification manager
  NotificationManager.initialize()
}

/**
 * Prepare and initialize the express api server.
 */
export function initializeApiServer () {
  const app = express()

  // Set system default time zone to UTC
  LuxonSettings.throwOnInvalid = true
  LuxonSettings.defaultZoneName = 'utc'

  // Set up body parsers middlewares
  app.use(express.json())
  app.use(express.urlencoded({ extended: true }))

  // Set up morgan logger middleware
  app.use(morgan('combined', {
    stream: rfs.createStream('access.log', {
      interval: '1d', // rotate daily
      path: process.env.LOGS_FOLDER_PATH,
      compress: true
    })
  }))
  app.use(morgan(process.env.NODE_ENV === 'production' ? 'tiny' : 'dev', {
    stream: { write: msg => debugApi(msg.trimEnd()) }
  }))

  // Set up auth passport middleware
  app.use(passport.initialize()) // Starts passport
  passport.use(new StrategyJwt(
    {
      secretOrKey: process.env.AUTH_SECRET,
      jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
      issuer: process.env.AUTH_ISSUER,
      algorithms: process.env.AUTH_ALGORITHM,
      ignoreExpiration: process.env.NODE_ENV === 'development',
      passReqToCallback: false
    },
    function (jwtPayload, done) {
      done(null, jwtPayload.user.id ?? false)
    }
  ))

  // Set up i18n middlewares
  i18n.configure({
    locales: process.env.SUPPORTED_LOCALES.split(','),
    defaultLocale: process.env.DEFAULT_LOCALE,
    retryInDefaultLocale: true,
    directory: path.join(__dirname, 'locales'),
    updateFiles: false,
    objectNotation: true
  })
  app.use(i18n.init)

  app.use(new I18nMongooseError().handler(
    function (err, req, res, next) {
      res.status(StatusCodes.UNPROCESSABLE_ENTITY).json(err)
    })
  )

  // Set up luxon settings middleware
  app.use(function (req, res, next) {
    LuxonSettings.defaultLocale = req.locale ?? process.env.DEFAULT_LOCALE
    next()
  })

  // Set up others middlewares
  app.use(helmet())
  app.use(cors())

  // Set static routes
  app.use('/upload', express.static(process.env.UPLOADS_FOLDER_PATH))

  // Service status checking route
  app.get('/', function (req, res) {
    res.status(StatusCodes.OK).send('Service is up and running')
  })

  // Set api routes
  const router = express.Router()

  router.use('/debug', DebugRouter) // Auth router
  router.use('/auth', AuthRouter) // Auth router
  router.use('/user', UserRouter) // User router

  app.use('/api', router)

  // Catch 404
  app.use((req, res, next) => {
    const err = new ApiError(ReasonPhrases.NOT_FOUND, StatusCodes.NOT_FOUND)
    next(err)
  })

  // Custom error handler
  app.use(function (err, req, res, next) {
    // Ignore if headers already sent
    if (res.headersSent) {
      return next(err)
    }

    debugApi('%O', err)

    let status = StatusCodes.INTERNAL_SERVER_ERROR
    if (err.status) {
      status = err.status
    } else if (err instanceof mongoose.Error.ValidationError) {
      status = StatusCodes.UNPROCESSABLE_ENTITY
    }

    res.status(status)
    res.json({
      message: err.message,
      errors: err.errors
    })
  })

  // Server listen
  app.listen(process.env.API_PORT, function () {
    var host = this.address().address === '::' ? 'localhost' : this.address().address
    var port = this.address().port

    debugApp(
      'CarPooling4IPS Back-end API is up and listening at http://%s:%s',
      host,
      port
    )
  })
}
