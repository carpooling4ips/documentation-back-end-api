'use strict'

/*

import express from 'express'
import expressValidator from 'express-validator'
import HttpStatus from 'http-status-codes'
import mongoose from 'mongoose'

import ApiError from '../../utils/apiError.js'
import Validator from '../../utils/validator.js'
import { VehicleModel, UserModel, RideGroupModel } from '../../models/index.js'

/*
 * ==================================================
 * | Constants                                      |
 * ==================================================

const { checkSchema, validationResult } = expressValidator
const { ReasonPhrases, StatusCodes } = HttpStatus

/*

/**

  /*

  /**
   * Validate get vehicle detail input.
   *
   * @returns {object[]} Array of validator functions.
   */
  static validateGetVehicleDetail () {
    return [
      checkSchema({
        vehicleId: {
          in: ['params'],
          isMongoId: true
        }
      })
    ]
  }

  /**
   * Validate create vehicle input.
   *
   * @returns {object[]} Array of validator functions.
   */
  static validateCreateVehicle () {
    return [
      checkSchema({
        alias: {
          in: ['body'],
          isString: true,
          trim: true,
          escape: true
        },
        fuelType: {
          in: ['body'],
          isInt: {
            options: {
              min: 1,
              max: 4
            }
          },
          toInt: true
        },
        consumption: {
          in: ['body'],
          isFloat: {
            options: {
              min: 0
            }
          },
          toFloat: true
        },
        emissions: {
          in: ['body'],
          custom: {
            options: (value, { req, location, path }) => {
              if (req.body.fuelType === 4) return true

              if (typeof value === 'number') return value > 0
              return Validator.isFloat(value, { min: 0 })
            }
          },
          customSanitizer: {
            options: (value, { req, location, path }) => {
              if (req.body.fuelType === 4) return undefined

              if (typeof value === 'number') return value

              const emissions = Validator.toFloat(value)
              if (isNaN(emissions)) return undefined
              return emissions
            }
          }
        },
        seats: {
          in: ['body'],
          isInt: {
            options: {
              min: 1,
              max: 8
            }
          },
          toInt: true
        }
      })
    ]
  }

  /**
   * Get the user's vehicles.
   *
   * @param {express.Request}       req   The express request object.
   * @param {express.Response}      res   The express response object.
   * @param {express.NextFunction}  next  The express next middleware function.
   */
  static async getVehicles (req, res, next) {
    try {
      // Check the logged-in user permission
      if (req.params.user !== 'me') {
        throw new ApiError(ReasonPhrases.FORBIDDEN, StatusCodes.FORBIDDEN)
      }

      // Set request user's id
      const userId = req.user

      // Check for request validation errors
      const errors = validationResult(req)
      if (!errors.isEmpty()) {
        throw new ApiError('Request fields are invalid!', StatusCodes.UNPROCESSABLE_ENTITY, errors.array()) // TODO Use i18n
      }

      // Prepare find conditions
      const findConditions = {
        isDisabled: false,
        _id: userId
      }

      // Retrieve ride vehicle documents
      const vehicleDocs = await UserModel.findOne(findConditions, {
        vehicles: true
      })
        .populate('vehicles', 'alias fuelType consumption emissions seats')
        .map(doc => {
          return doc.vehicles.map(vehicle => {
            return {
              id: vehicle._id,
              alias: vehicle.alias,
              fuelType: vehicle.fuelType,
              consumption: vehicle.consumption,
              emissions: vehicle.emissions,
              seats: vehicle.seats
            }
          })
        })
        .exec()

      res.status(200).send(vehicleDocs)
    } catch (err) {
      next(err)
    }
  }

  /**
   * Get the specified vehicle details.
   *
   * @param {express.Request}       req   The express request object.
   * @param {express.Response}      res   The express response object.
   * @param {express.NextFunction}  next  The express next middleware function.
   */
  static async getVehicleDetail (req, res, next) {
    try {
      // Check the logged-in user permission
      if (req.params.user !== 'me') {
        throw new ApiError(ReasonPhrases.FORBIDDEN, StatusCodes.FORBIDDEN)
      }

      // Check for request validation errors
      const errors = validationResult(req)
      if (!errors.isEmpty()) {
        throw new ApiError('Request fields are invalid!', StatusCodes.UNPROCESSABLE_ENTITY, errors.array()) // TODO Use i18n
      }

      // Set request user's id
      const userId = req.user
      // Prepare find conditions
      const findConditions = {
        isDisabled: false,
        vehicles: { $in: [req.params.vehicleId] },
        _id: userId
      }
      // Retrieve specified vehicle document
      const userDocs = await UserModel.findOne(findConditions, {
        vehicles: true
      })
        .populate('vehicles', 'alias fuelType consumption emissions seats')
        .map(doc => {
          doc.vehicles = doc.vehicles.filter(vehicle => {
            return vehicle._id.equals(req.params.vehicleId)
          })
          return doc
        })
        .exec()

      const vehicleDoc = userDocs.vehicles[0]

      if (!vehicleDoc) {
        throw new ApiError(ReasonPhrases.NOT_FOUND, StatusCodes.NOT_FOUND)
      }

      res.status(StatusCodes.OK).send(vehicleDoc)
    } catch (err) {
      next(err)
    }
  }

  /**
   * Create a new vehicle for the specified user.
   *
   * @param {express.Request}       req   The express request object.
   * @param {express.Response}      res   The express response object.
   * @param {express.NextFunction}  next  The express next middleware function.
   */
  static async createVehicle (req, res, next) {
    try {
      // Check the logged-in user permission
      if (req.params.user !== 'me') {
        throw new ApiError(ReasonPhrases.FORBIDDEN, StatusCodes.FORBIDDEN)
      }

      // Set request user's id
      const userId = req.user

      // Check for request validation errors
      const errors = validationResult(req)
      if (!errors.isEmpty()) {
        throw new ApiError('Request fields are invalid!', StatusCodes.UNPROCESSABLE_ENTITY, errors.array()) // TODO Use i18n
      }

      const user = await UserModel.findById(userId).populate('vehicles', 'alias fuelType consumption emissions seats').exec()

      // check if the user already has a vehicle with the same alias
      // TODO Move this check to the model
      if (user.vehicles.filter(v => v.alias === req.body.alias).length > 0) {
        throw new ApiError(ReasonPhrases.CONFLICT, StatusCodes.CONFLICT)
      }
      console.log(req.body.emissions)
      const vehicle = new VehicleModel({
        alias: req.body.alias,
        fuelType: req.body.fuelType,
        consumption: req.body.consumption,
        emissions: req.body.emissions,
        seats: req.body.seats
      })

      // TODO ADD TRANSACTION
      // create the vehicle and add it to the user's document after
      await vehicle.save().then(async doc => {
        await UserModel.updateOne({ _id: userId }, { $push: { vehicles: doc._id } }).exec()
      })
      // Send response to the user
      res.sendStatus(StatusCodes.CREATED)
    } catch (err) {
      next(err)
    }
  }

  /**
   * Edit the specified ride group.
   *
   * @param {express.Request}       req   The express request object.
   * @param {express.Response}      res   The express response object.
   * @param {express.NextFunction}  next  The express next middleware function.
   */
  static async editVehicle (req, res, next) {
    try {
      // Check the logged-in user permission
      if (req.params.user !== 'me') {
        throw new ApiError(ReasonPhrases.FORBIDDEN, StatusCodes.FORBIDDEN)
      }

      // Set request user's id
      const userId = req.user

      // Check for request validation errors
      const errors = validationResult(req)
      if (!errors.isEmpty()) {
        throw new ApiError('Request fields are invalid!', StatusCodes.UNPROCESSABLE_ENTITY, errors.array()) // TODO Use i18n
      }

      const userDoc = await UserModel.findOne({
        _id: userId
      }).populate('vehicles', 'alias fuelType consumption emissions seats').exec()

      // check if user has this vehicle or if his vehicles already have the same alias provided
      if (userDoc.vehicles.filter(v => v._id.equals(req.params.vehicleId)).length === 0) {
        throw new ApiError('The vehicle does not belong to the user!', StatusCodes.CONFLICT)
      }

      if (
        userDoc.vehicles.filter(v => v.alias.toUpperCase() === req.body.alias.toUpperCase() &&
          !v._id.equals(req.params.vehicleId)).length > 0) {
        throw new ApiError('A vehicle with the same name already exists!', StatusCodes.CONFLICT)
      }

      // Retrieve the vehicle group
      const vehicleDoc = await VehicleModel.findOne({
        _id: req.params.vehicleId
      }).exec()

      if (!vehicleDoc) {
        throw new ApiError(ReasonPhrases.NOT_FOUND, StatusCodes.NOT_FOUND)
      }

      // Update the vehicle
      vehicleDoc.alias = req.body.alias

      // TODO MAYBE UPDATE SEATS?? NOT UPDATING DUE TO RIDE GROUPS CONFLICTS
      await vehicleDoc.save()

      // Send response to the user
      res.sendStatus(StatusCodes.NO_CONTENT)
    } catch (err) {
      next(err)
    }
  }

  /**
   * Delete the specified vehicle.
   *
   * @param {express.Request}       req   The express request object.
   * @param {express.Response}      res   The express response object.
   * @param {express.NextFunction}  next  The express next middleware function.
   */
  static async deleteVehicle (req, res, next) {
    try {
      // Check the logged-in user permission
      if (req.params.user !== 'me') {
        throw new ApiError(ReasonPhrases.FORBIDDEN, StatusCodes.FORBIDDEN)
      }

      // Set request user's id
      const userId = req.user

      // Check for request validation errors
      const errors = validationResult(req)
      if (!errors.isEmpty()) {
        throw new ApiError('Request fields are invalid!', StatusCodes.UNPROCESSABLE_ENTITY, errors.array()) // TODO Use i18n
      }

      const userDoc = await UserModel.findOne({
        _id: userId,
        isDisabled: false
      }).populate('vehicles', 'alias fuelType consumption emissions seats')
        .exec()

      if (userDoc.vehicles.filter(v => v._id.equals(req.params.vehicleId)).length === 0) {
        throw new ApiError('The vehicle does not belong to the current user!', StatusCodes.NOT_FOUND)
      }

      // Retrieve ride group alias and passengers
      const vehicleDoc = await VehicleModel.findOne({
        _id: req.params.vehicleId
      }).exec()

      if (!vehicleDoc) {
        throw new ApiError('The vehicle does not exist!', StatusCodes.NOT_FOUND)
      }

      const rideGroup = await RideGroupModel.findOne({
        vehicleId: req.params.vehicleId,
        isDisabled: false
      }).lean().exec()

      // check if vehicle is part of an active ride group
      if (rideGroup) {
        throw new ApiError('The vehicle is being used in a ride group!', StatusCodes.PRECONDITION_REQUIRED)
      }

      // Start a mongodb transaction session
      await mongoose.connection.transaction(async (session) => {
        // Disable the vehicle
        // NOTE: This query does not trigger mongoose schema validators

        const userDocument = await UserModel.findOne({
          _id: userId,
          isDisabled: false
        })
          .exec()

        await VehicleModel.deleteOne({
          _id: req.params.vehicleId
        }, {
          session: session
        }).exec()

        // pull vehicle id from user vehicles
        // NOTE: This query does not trigger mongoose schema validators
        userDocument.vehicles = userDocument.vehicles.filter(v => !v._id.equals(req.params.vehicleId))
        await userDocument.save()
      })

      // TODO NOTIFY USERS OF VEHICLE DELETION
      // Notify the passengers that the ride group has been deleted
      /* NotificationManager.sendAll(rideGroupDoc.passengers, 'PASSENGER_RIDE_GROUP_DELETED', {
        RIDE_GROUP_ALIAS: rideGroupDoc.alias
      }) */

      // Send response to the user
      res.sendStatus(StatusCodes.NO_CONTENT)
    } catch (err) {
      next(err)
    }
  }
