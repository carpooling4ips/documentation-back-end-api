src/routes/debug/debug.notification.js

/**
 * @module DebugNotificationRoute
 * @category Routes
 */

'use strict'

/*
 * ==================================================
 * | Imports                                        |
 * ==================================================
 */
import express from 'express'

// Controllers
import { DebugNotificationController } from '../../controllers/index.js'

/*
 * ==================================================
 * | Constants                                      |
 * ==================================================
 */
const router = express.Router({
  mergeParams: true
})

/*
 * ==================================================
 * | Routes                                         |
 * ==================================================
 */

/**
 * Notify specified user.
 *
 * @name Debug Notify User
 * @path {POST} /debug/notification
 * @body {string}   userId              The user id to send the debug notification.
 * @body {string}   notificationKey     The notification key to send.
 * @body {string}   notificationParams  The specified notification parameters.
 * @body {string}   notificationData    The specified notification extra data.
 * @code {204} The debug notification sent to the user id succeeded.
 * @code {422} The request content is unable to be processed due to invalid fields.
 * @code {500} An internal server error.
 * @example <caption>Possible input:</caption>
 * {
 *   "userId": "5f918a6b75975c50841391fc",
 *   "notificationKey": "PASSENGER_REMOVED_RIDE_GROUP",
 *   "notificationParams": {
 *     "RIDE_GROUP_ALIAS": "Test Group 1"
 *   },
 *   "notificationData": {
 *     "rideGroupId": "5ff12e97949c34545059e634"
 *   }
 * }
 */
router.post('/', DebugNotificationController.validateNotifyUser(), DebugNotificationController.notifyUser)

/**
 * The router-layer of the debug notification.
 */
export default router
