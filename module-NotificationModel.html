Example

Example usage of Notification model.

let notification1 = new Notification({
  key: "NEW_MESSAGE",
  title: {
    pt: "Nova mensagem!",
    en: "New message!"
  },
  body: {
    pt: "Nova mensagem!",
    en: "New message!"
  },
  type: 2,
  data: {
    routeName: "/newRouteName"
  }
});
