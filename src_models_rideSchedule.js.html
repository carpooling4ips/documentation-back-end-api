/**
 * @module RideScheduleModel
 * @category Models
 */

'use strict'

/*
 * ==================================================
 * | Imports                                        |
 * ==================================================
 */
import mongoose from 'mongoose'

// Models
import { RideScheduleModel, RideGroupModel } from './index.js'

/**
 * Enum for ride schedule passenger request status.
 *
 * @typedef {PassengerRequestStatus} PassengerRequestStatus
 * @readonly
 * @property {number}   Pending   Pending passenger ride request.
 * @property {number}   Accepted  Accepted passenger ride request
 * @property {number}   Rejected  Rejected passenger ride request.
 */
const PassengerRequestStatus = Object.freeze({
  Pending: 1,
  Accepted: 2,
  Rejected: 3,
  Canceled: 4
})

/**
 * Enum for ride schedule status.
 *
 * @typedef {Status} Status
 * @readonly
 * @property {number}   Schedule            Scheduled ride.
 * @property {number}   WaitingPassengers   Waiting for passengers at the meeting point.
 * @property {number}   Started             Started ride to the destination.
 * @property {number}   Finished            Finished ride.
 * @property {number}   Canceled            Canceled ride.
 */
const Status = Object.freeze({
  Scheduled: 1,
  WaitingPassengers: 2,
  Started: 3,
  Finished: 4,
  Canceled: 5
})

/*
 * ==================================================
 * | Schemas                                        |
 * ==================================================
 */

/**
 * Passenger request nested schema.
 *
 * @type {mongoose.Schema}
 * @property {mongoose.ObjectId}        userId          Passenger's user id.
 * @property {Date}                     [handlingDate]  Passenger ride handling (approve or refuse) date.
 * @property {PassengerRequestStatus}   [status]        Passenger ride request approval.
 * @property {boolean}                  [isDisabled]    Passenger status (true if deleted).
 */
const PassengerRequestSchema = new mongoose.Schema({
  userId: {
    type: mongoose.ObjectId,
    ref: 'User',
    validate: async function (value) {
      const passengerAlreadyExists = await RideScheduleModel.exists({ 'passengerRequests._id': this._id })
      if (passengerAlreadyExists) {
        return true
      }

      // Validate if the passenger is already part of the parent ride group
      const passengerPartOfRideGroup = await RideGroupModel.exists({
        _id: this.parent().rideGroupId,
        ['passengers.' + value]: { $exists: true }
      })
      if (passengerPartOfRideGroup) {
        throw new Error('This user account is already part of the ride group passengers') // TODO Use i18n
      }

      // Validate if the passenger already has an active request
      const passengerActiveRequest = await RideScheduleModel.exists({
        _id: this.parent()._id,
        passengerRequests: {
          $elemMatch: {
            _id: { $ne: this._id },
            userId: value,
            status: { $in: [PassengerRequestStatus.Accepted, PassengerRequestStatus.Pending] },
            isDisabled: false
          }
        }
      })
      if (passengerActiveRequest) {
        throw new Error('This user account already has an accepted or pending ride request for the specified schedule') // TODO Use i18n
      }

      return true
    },
    required: true
  },
  handlingDate: { type: Date },
  status: { type: Number, enum: Object.values(PassengerRequestStatus), required: true, default: PassengerRequestStatus.Pending },
  isDisabled: { type: Boolean, select: false, required: true, default: false }
}, {
  timestamps: true
})

PassengerRequestSchema.index({ isDisabled: 1, isApproved: 1 })
PassengerRequestSchema.index({ isDisabled: 1, userId: 1 })

/**
 * Confirmed passenger nested schema.
 *
 * @type {mongoose.Schema}
 * @property {mongoose.ObjectId}  userId  Passenger's user id.
 */
const ConfirmedPassengerSchema = new mongoose.Schema({
  userId: {
    type: mongoose.ObjectId,
    ref: 'User',
    validate: async function (value) {
      // Check if the passenger is part of the ride group
      const passengerPartOfRideGroup = await RideGroupModel.exists({
        _id: this.parent().rideGroupId,
        ['passengers.' + value]: { $exists: true }
      })

      // If is not part of the ride group check if the passenger has an accepted request to participate on the ride schedule
      if (!passengerPartOfRideGroup) {
        const passengerPartOfRideSchedule = await RideScheduleModel.exists({
          _id: this.parent()._id,
          passengerRequests: {
            $elemMatch: {
              userId: value,
              status: PassengerRequestStatus.Accepted,
              isDisabled: false
            }
          }
        })
        if (!passengerPartOfRideSchedule) {
          throw new Error('This user account is not a passenger of the specified ride schedule nor of the parent ride group') // TODO Use i18n
        }
      }

      return true
    },
    required: true
  }
}, {
  timestamps: true
})

export { ConfirmedPassengerSchema }

/**
 * Ride schedule schema.
 *
 * @type {mongoose.Schema}
 * @property {mongoose.ObjectId}          rideGroupId             Ride schedule's group id.
 * @property {Date}                       realizationDate         Ride schedule realization date and time.
 * @property {PassengerRequestSchema[]}   [passengerRequests]     Ride schedule's passengers which requested to participate in the ride.
 * @property {mongoose.ObjectId[]}        [confirmedPassengers]   Ride schedule's passengers which confirmed their presence in the ride.
 * @property {Status}                     [status]                Ride schedule status.
 * @property {boolean}                    [isDisabled]            Ride schedule status (true if deleted).
 */
const RideScheduleSchema = new mongoose.Schema({
  rideGroupId: { type: mongoose.ObjectId, ref: 'RideGroup', index: true, required: true },
  realizationDate: { type: Date, required: true },
  passengerRequests: { type: [PassengerRequestSchema] },
  confirmedPassengers: { type: Map, of: ConfirmedPassengerSchema },
  status: { type: Number, enum: Object.values(Status), required: true, default: Status.Scheduled },
  isDisabled: { type: Boolean, select: false, required: true, default: false }
}, {
  timestamps: true
})

RideScheduleSchema.index({ isDisabled: 1, status: 1, realizationDate: -1 })

// Define schema statics
Object.assign(RideScheduleSchema.statics, {
  Status,
  PassengerRequestStatus
})

/**
 * The data-layer for a ride schedule.
 *
 * @example <caption>Example usage of RideSchedule model.</caption>
 * let rideSchedule1 = new RideSchedule({
 *   rideGroupId: mongoose.Types.ObjectId('5f574fe68b17b9087e60ac0c'),
 *   realizationDate: new Date(),
 *   passengerRequests: [{
 *     userId: mongoose.Types.ObjectId('5f58f5876da0db3ca0955318'),
 *     handlingDate: new Date(),
 *     status: RideSchedule.schema.statics.PassengerRequestStatus.Accepted
 *   }],
 *   confirmedPassengers: [{
 *     userId: mongoose.Types.ObjectId('5f58f4ea9207efc18929c52a')
 *   }],
 *   status: RideSchedule.schema.statics.Status.Finished
 * });
 */
export default mongoose.model('RideSchedule', RideScheduleSchema)
