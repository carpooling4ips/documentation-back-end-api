src/models/news.js

/**
 * @module NewsModel
 * @category Models
 */

'use strict'

/*
 * ==================================================
 * | Imports                                        |
 * ==================================================
 */
import mongoose from 'mongoose'

/*
 * ==================================================
 * | Schemas                                        |
 * ==================================================
 */

/**
 * News schema.
 *
 * @type {mongoose.Schema}
 * @property {string}   title         News title.
 * @property {string}   description   News description.
 * @property {string}   [image]         News image file name on system.
 * @property {boolean}  [isDisabled]  News status (true if deleted).
 */
const NewsSchema = new mongoose.Schema({
  title: { type: String, minlength: 3, maxlength: 64, trim: true, required: true },
  description: { type: String, minlength: 3, maxlength: 4096, trim: true, required: true },
  image: { type: String },
  isDisabled: { type: Boolean, select: false, required: true, default: false }
}, {
  timestamps: true
})

NewsSchema.index({ isDisabled: 1, date: -1 })

/**
 * The data-layer for a news.
 *
 * @example <caption>Example usage of News model.</caption>
 * let news1 = new News({
 *   title: "News 1",
 *   description: "This is a new news.\nIPS forever!",
 *   image: '5f564e71cb0606d26012a2d5'
 * });
 */
export default mongoose.model('News', NewsSchema)
