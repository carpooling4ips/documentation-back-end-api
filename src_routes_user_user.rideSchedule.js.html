/**
 * Get all user's ride schedules.
 *
 * @name Ride Schedule Get All
 * @path {GET} /user/:user/rideSchedule
 * @auth This route requires a valid user's token. If authentication fails it will return a 401 http status code.
 * @header {string}   Authorization   The authentication user's token.
 * @params {string}   :user   Use 'me' to reference the logged-in user or specify a user's id to act on the behalf of that user.
 * @query {string}  mode      Specifies if retrieving user's ride schedules as a driver or as a passenger. Possible values: driver, passenger.
 * @query {number}  [week]    Specifies the week number of the week year to retrieve the ride schedules. Possible values: 1-53. At least one week or month must be specified, if both specified, the month take precedence.
 * @query {number}  [month]   Specifies the month number to retrieve the ride schedules. Possible values: 1-12. At least one week or month must be specified, if both specified, the month take precedence.
 * @query {number}  year      Specifies year number to retrieve the ride schedules.
 * @code {200} The user's ride schedules retrieval succeeded.
 * @code {401} The user authentication has failed.
 * @code {403} The authenticated user does not have permission.
 * @code {422} The request content is unable to be processed due to invalid fields.
 * @code {500} An internal server error.
 * @response {string}     id                The ride schedule's id.
 * @response {string}     alias             The ride schedule's alias
 * @response {string}     realizationDate   The ride schedule's realization date (iso 8601 format).
 * @response {number}     status            The ride schedule's status. Possible values: 1 (schedule), 2 (waiting for passengers), 3 (started), 4 (finished), 5 (canceled).
 * @example <caption>Possible request:</caption>
 * http://127.0.0.1:2450/api/user/me/rideSchedule?mode=driver&month=11&year=2020
 * @example <caption>Possible output:</caption>
 * [
 *   {
 *     "id": "5fa2a0c8481a0e3bd017c21d",
 *     "alias": "Test 2",
 *     "realizationDate": "2020-11-05T13:30:00.000Z",
 *     "status": 1
 *   }
 * ]
 */
router.get('/', passport.authenticate('jwt', { session: false }), UserRideScheduleController.validateGetRideSchedules(), UserRideScheduleController.getRideSchedules)
