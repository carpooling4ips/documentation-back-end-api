/**
 * @module UserNotificationModel
 * @category Models
 */

'use strict'

/*
 * ==================================================
 * | Imports                                        |
 * ==================================================
 */
import mongoose from 'mongoose'

import Validator from '../utils/validator.js'

/*
 * ==================================================
 * | Schemas                                        |
 * ==================================================
 */

/**
 * User notification schema.
 *
 * @type {mongoose.Schema}
 * @property {mongoose.ObjectId}  userId          User id which received the notification.
 * @property {mongoose.ObjectId}  notificationId  Notification template id.
 * @property {Map<string>}        [params]        Custom parameters to fill notification template.
 * @property {string}             locale          User received notification locale.
 * @property {string}             rawMessage      User received raw notification with parameters filled.
 * @property {string}             fcmMessageCode  Firebase cloud messaging code assigned to the sent notification.
 * @property {boolean}            [isRead]        If notification has been seen by the user.
 */
const UserNotificationSchema = new mongoose.Schema({
  userId: { type: mongoose.ObjectId, ref: 'User', required: true },
  notificationId: { type: mongoose.ObjectId, ref: 'Notification', required: true },
  params: { type: Map, of: String },
  locale: {
    type: String,
    validate: {
      validator: Validator.isLocale,
      message: '"{VALUE}" is not a valid locale' // TODO Use i18n
    },
    required: true
  },
  rawMessage: { type: String, required: true },
  fcmMessageCode: { type: String, required: true },
  isRead: { type: Boolean, required: true, default: false }
}, {
  timestamps: true
})

/**
 * The data-layer for a user's notification.
 *
 * @example <caption>Example usage of UserNotification model.</caption>
 * let userNotification1 = new UserNotification({
 *   userId: mongoose.Types.ObjectId('5f490f9ae034e1f255ee6653'),
 *   notificationId: mongoose.Types.ObjectId('5f490fcd5b57f76ec45c21cf'),
 *   params: {
 *     VAR1: 'Edgar',
 *     VAR2: 'Santos'
 *   },
 *   locale: "pt",
 *   rawMessage: "First name: Edgar; Last name: Santos.",
 *   fcmMessageCode: "projects/myproject-b5ae1/messages/0:1500415314455276%31bd1c9631bd1c96"
 * });
 */
export default mongoose.model('UserNotification', UserNotificationSchema)
