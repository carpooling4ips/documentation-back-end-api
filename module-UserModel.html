Example

Example usage of User model.

let user1 = new User({
  aadId: '5f5a506c305ff6bd4475h3s6',
  name: 'User1',
  surname: 'Surname1',
  title: 'Student',
  email: 'user1@email.com',
  avatar: {
    fileName: '2593ee56cc8f8bb16f13995e4f18b746',
    originalName: 'WIN_20200729_09_52_41_Pro.jpg',
    mimeType: 'image/jpeg',
    size: 121976
  },
  vehicles: [
    mongoose.Types.ObjectId('5f5a506c305ff6bd4475b1d1'),
    mongoose.Types.ObjectId('5f5a506fc35e608a4832f569')
  ],
  level: 4,
  experience: 2342,
  blockedUsers: {
    '5f5b444fb804793cf3faed87': {
      userId: mongoose.Types.ObjectId('5f5b444fb804793cf3faed87')
    }
  }
});
