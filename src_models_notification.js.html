/**
 * @module NotificationModel
 * @category Models
 */

'use strict'

/*
 * ==================================================
 * | Imports                                        |
 * ==================================================
 */
import mongoose from 'mongoose'
import mongooseIntl from 'mongoose-intl'

/*
 * ==================================================
 * | Constants                                      |
 * ==================================================
 */

/**
 * Enum for notification types.
 *
 * @typedef {NotificationTypes} NotificationTypes
 * @readonly
 * @property {number}   Info    Just a info notification with no action.
 * @property {number}   Route   Notification which opens an app route on click.
 */
const NotificationTypes = Object.freeze({
  Info: 1,
  Route: 2,
  Navigate: 3
})

/*
 * ==================================================
 * | Schemas                                        |
 * ==================================================
 */

/**
 * Notification schema.
 *
 * @type {mongoose.Schema}
 * @property {string}             key     Notification unique key.
 * @property {string}             title   Notification title (multilingual).
 * @property {string}             body    Notification message (multilingual).
 * @property {NotificationTypes}  type    Notification type.
 * @property {Map<string>}        data    Notification extra data.
 */
const NotificationSchema = new mongoose.Schema({
  key: { type: String, unique: true, minlength: 3, maxlength: 64, uppercase: true, trim: true, required: true },
  title: { type: String, minlength: 3, maxlength: 64, required: true, intl: true },
  body: { type: String, minlength: 3, maxlength: 400, required: true, intl: true },
  type: { type: Number, enum: Object.values(NotificationTypes), required: true },
  data: { type: Map, of: String }
})

Object.assign(NotificationSchema.statics, {
  NotificationTypes
})

// Set schema plugins
NotificationSchema.plugin(mongooseIntl, {
  languages: process.env.SUPPORTED_LOCALES.split(','),
  defaultLanguage: process.env.DEFAULT_LOCALE
})

/**
 * The data-layer for a notification.
 *
 * @example <caption>Example usage of Notification model.</caption>
 * let notification1 = new Notification({
 *   key: "NEW_MESSAGE",
 *   title: {
 *     pt: "Nova mensagem!",
 *     en: "New message!"
 *   },
 *   body: {
 *     pt: "Nova mensagem!",
 *     en: "New message!"
 *   },
 *   type: 2,
 *   data: {
 *     routeName: "/newRouteName"
 *   }
 * });
 */
export default mongoose.model('Notification', NotificationSchema)
