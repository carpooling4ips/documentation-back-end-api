/**
 * @module UserProfileController
 * @category Controllers
 */

'use strict'

/*
 * ==================================================
 * | Imports                                        |
 * ==================================================
 */
import express from 'express'
import expressValidator from 'express-validator'
import HttpStatus from 'http-status-codes'
import luxon from 'luxon'
import path from 'path'
import fs from 'fs-extra'
import debug from 'debug'

import ApiError from '../../utils/apiError.js'

// Models
import { UserModel, RideHistoryModel } from '../../models/index.js'

/*
 * ==================================================
 * | Constants                                      |
 * ==================================================
 */
const { checkSchema, validationResult } = expressValidator
const { ReasonPhrases, StatusCodes } = HttpStatus
const { DateTime } = luxon

// Debug package
const debugApp = debug('carpooling4ips')
const debugFS = debugApp.extend('fs')

/*
 * ==================================================
 * | Exports                                        |
 * ==================================================
 */

/**
 * Class representing the user's profile controller.
 *
 * @category Controllers
 */
export default class {
  /*
  * ==================================================
  * | Public validation functions                    |
  * ==================================================
  */

  /**
   * Validate edit profile avatar route input.
   *
   * @returns {object[]} Array of validator functions.
   */
  static validateEditProfileAvatar () {
    return [
      checkSchema({
        avatar: {
          in: ['body'],
          custom: {
            options: (value, { req, location, path }) => {
              return !!req.file
            },
            errorMessage: 'You should upload at least one avatar image file up to 1Mb' // TODO Use i18n
          }
        }
      })
    ]
  }

  /*
  * ==================================================
  * | Public route functions                         |
  * ==================================================
  */

  /**
   * Get specified user's profile data.
   *
   * @param {express.Request}       req   The express request object.
   * @param {express.Response}      res   The express response object.
   * @param {express.NextFunction}  next  The express next middleware function.
   */
  static async getProfile (req, res, next) {
    try {
      // Set request user's id
      const userId = req.params.user === 'me' ? req.user : req.params.user

      // Retrieve the user data from database
      const promiseUser = UserModel.findOne({
        _id: userId,
        isSuspended: false,
        isDisabled: false
      })
        .lean({ virtuals: true })
        .exec()
        .then(function (doc) {
          if (!doc) {
            throw new ApiError(ReasonPhrases.NOT_FOUND, StatusCodes.NOT_FOUND)
          }

          return doc
        })

      // Retrieve user's asked rides count
      const promiseAskedRides = RideHistoryModel.countDocuments({
        ['passengers.' + userId]: { $exists: true }
      }).exec().then(function (count) {
        return count
      })

      // Retrieve user's given rides and process rating average and comments
      const promiseGivenRides = RideHistoryModel.find({
        driverId: userId
      }, {
        passengers: 1
      })
        .lean()
        .exec()
        .then(function (docs) {
          const givenRidesData = {
            count: docs.length,
            ratingSum: 0,
            ratingCount: 0,
            comments: []
          }

          for (const givenRide of docs) {
            if (!givenRide.passengers) continue

            for (const passenger of Object.values(givenRide.passengers)) {
              if (!passenger.rating) continue

              givenRidesData.ratingSum += passenger.rating.rating
              ++givenRidesData.ratingCount

              if (passenger.rating.comment) {
                givenRidesData.comments.push({
                  text: passenger.rating.comment,
                  date: DateTime.fromJSDate(passenger.rating.createdAt).toUTC().toISO()
                })
              }
            }
          }

          return givenRidesData
        })

      // Retrieve if logged-in user blocked retrieving user
      let promiseIsBlocked
      if (req.user !== userId) {
        promiseIsBlocked = UserModel.countDocuments({
          _id: req.user,
          ['blockedUsers.' + userId]: { $exists: true },
          isSuspended: false,
          isDisabled: false
        }).exec().then(function (count) {
          return !!count
        })
      }

      // Execute promises
      const [userDoc, askedRidesCount, givenRidesData, isBlocked] = await Promise.all([promiseUser, promiseAskedRides, promiseGivenRides, promiseIsBlocked])

      // Return user's profile data
      const userProfile = {
        name: userDoc.name,
        title: userDoc.title,
        avatar: userDoc.avatar,
        rating: givenRidesData.ratingCount ? Math.round((givenRidesData.ratingSum / givenRidesData.ratingCount) * 2) / 2 : 0,
        ratingCount: givenRidesData.ratingCount,
        askedRides: askedRidesCount,
        givenRides: givenRidesData.count,
        totalAchievements: 0, // TODO Add total achievements
        comments: givenRidesData.comments,
        isBlocked: isBlocked
      }
      res.status(StatusCodes.OK).send(userProfile)
    } catch (err) {
      next(err)
    }
  }

  /**
   * Edit specified user's profile avatar image.
   *
   * @param {express.Request}       req   The express request object.
   * @param {express.Response}      res   The express response object.
   * @param {express.NextFunction}  next  The express next middleware function.
   */
  static async editProfileAvatar (req, res, next) {
    try {
      // Check the logged-in user permission
      if (req.params.user !== 'me') {
        throw new ApiError(ReasonPhrases.FORBIDDEN, StatusCodes.FORBIDDEN)
      }

      // Set request user's id
      const userId = req.user

      // Check for request validation errors
      const errors = validationResult(req)
      if (!errors.isEmpty()) {
        throw new ApiError('Request fields are invalid!', StatusCodes.UNPROCESSABLE_ENTITY, errors.array()) // TODO Use i18n
      }

      // Set update query
      const update = {
        avatar: {
          fileName: req.file.filename,
          originalName: req.file.originalname,
          mimeType: req.file.mimetype,
          size: req.file.size
        }
      }

      // Update document on the database
      const doc = await UserModel.findOneAndUpdate({
        _id: userId,
        isSuspended: false,
        isDisabled: false
      }, update, { runValidators: true })
      if (!doc) {
        throw new ApiError(ReasonPhrases.NOT_FOUND, StatusCodes.NOT_FOUND)
      }

      // Remove old image file
      if (doc.avatar && doc.avatar.fileName) {
        const oldFilePath = path.join(process.env.UPLOADS_FOLDER_PATH, doc.avatar.fileName)
        fs.remove(oldFilePath)
          .then(function () {
            debugFS('The file \'%s\' has been removed from the file system.', oldFilePath)
          })
          .catch(function (err) {
            debugFS(err)
          })
      }

      // Send operation response
      res.sendStatus(StatusCodes.NO_CONTENT)
    } catch (err) {
      next(err)
    }
  }
}
