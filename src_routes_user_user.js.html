src/routes/user/user.js

/**
 * @module UserRoute
 * @category Routes
 */

'use strict'

/*
 * ==================================================
 * | Imports                                        |
 * ==================================================
 */
import express from 'express'
import passport from 'passport'

// Routes
import UserSettingsRouter from './user.settings.js'
import UserProfileRouter from './user.profile.js'
import UserBlacklistRouter from './user.userBlacklist.js'
import RideGroupRouter from './user.rideGroup.js'
import RideScheduleRouter from './user.rideSchedule.js'
import VehicleRouter from './user.vehicle.js'

// Controllers
import { UserController } from '../../controllers/index.js'

/*
 * ==================================================
 * | Constants                                      |
 * ==================================================
 */
const router = express.Router({
  mergeParams: true
})

/*
 * ==================================================
 * | Nested Routes                                  |
 * ==================================================
 */

router.use('/:user/settings', UserSettingsRouter)
router.use('/:user/profile', UserProfileRouter)
router.use('/:user/userBlacklist', UserBlacklistRouter)
router.use('/:user/rideGroup', RideGroupRouter)
router.use('/:user/rideSchedule', RideScheduleRouter)
router.use('/:user/vehicle', VehicleRouter)

/*
 * ==================================================
 * | Routes                                         |
 * ==================================================
 */

/**
 * Search for the specified user.
 *
 * @name User Search
 * @path {GET} /user/:user/search
 * @auth This route requires a valid user's token. If authentication fails it will return a 401 http status code.
 * @header {string}   Authorization   The authentication user's token.
 * @params {string}   :user   Use 'me' to reference the logged-in user or specify a user's id to act on the behalf of that user.
 * @query {string}  query   The user searching filter parameter string.
 * @code {200} The searching users retrieval succeeded.
 * @code {401} The user authentication has failed.
 * @code {403} The authenticated user does not have permission.
 * @code {404} The requested user id does not exist.
 * @code {422} The request content is unable to be processed due to invalid fields.
 * @code {500} An internal server error.
 * @response {string}   id                  The filtered user's id.
 * @response {string}   name                The filtered user's name.
 * @response {string}   title               The filtered user's title.
 * @response {string}   [avatar.fileName]   The filtered user's avatar file name on the server.
 * @response {string}   [avatar.mimeType]   The filtered user's avatar file mime type.
 * @example <caption>Possible request:</caption>
 * http://127.0.0.1:2450/api/user/me/search?query=Edgar
 * @example <caption>Possible output:</caption>
 * [
 *   {
 *     "id": "5f918a6b75975c50841391fc",
 *     "name": "Edgar Santos",
 *     "title": "Estudante",
 *     "avatar": {
 *       "fileName": "3229c0fae86ecf38929161777824f5c5",
 *       "mimeType": "image/jpeg"
 *     }
 *   }
 * ]
 */
router.get('/:user/search', passport.authenticate('jwt', { session: false }), UserController.validateSearchUser(), UserController.searchUser)

/**
 * The router-layer of the user.
 */
export default router
