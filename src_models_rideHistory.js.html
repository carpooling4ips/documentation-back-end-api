/**
 * @module RideHistoryModel
 * @category Models
 */

'use strict'

/*
 * ==================================================
 * | Imports                                        |
 * ==================================================
 */
import mongoose from 'mongoose'

/*
 * ==================================================
 * | Schemas                                        |
 * ==================================================
 */

/**
 * Rating nested schema.
 *
 * @type {mongoose.Schema}
 * @property {number}   rating      Rating starts from 0 to 5.
 * @property {string}   [comment]   Rating comment.
 */
const RatingSchema = new mongoose.Schema({
  rating: {
    type: Number,
    min: 0,
    max: 5,
    validate: {
      validator: function (value) {
        return value % 0.5 === 0
      },
      message: '"{VALUE}" is not a multiple of 0.5' // TODO Use i18n
    },
    required: true
  },
  comment: { type: String, minlength: 3, maxlength: 256, trim: true }
}, {
  timestamps: true
})

/**
 * Passenger nested schema.
 *
 * @type {mongoose.Schema}
 * @property {mongoose.ObjectId}  userId      Passenger's user id.
 * @property {boolean}            isPaid      Passenger already paid.
 * @property {RatingSchema}       rating    Passenger rating.
 */
const PassengerSchema = new mongoose.Schema({
  userId: { type: mongoose.ObjectId, ref: 'User', required: true },
  isPaid: { type: Boolean, required: true, default: false },
  rating: { type: RatingSchema }
}, {
  timestamps: true
})

/**
 * Ride history schema.
 *
 * @type {mongoose.Schema}
 * @property {mongoose.ObjectId}      rideScheduleId  Ride history's ride schedule id.
 * @property {mongoose.ObjectId}      vehicleId       Ride history vehicle id.
 * @property {mongoose.ObjectId}      driverId        Ride history driver id.
 * @property {Map<PassengerSchema>}   passengers      Ride history passengers.
 * @property {number}                 expectedCost    Ride history expected cost.
 * @property {number}                 [realCost]      Ride history real cost.
 * @property {Date}                   [startDate]     Ride history start date.
 * @property {Date}                   [endDate]       Ride history end date.
 * @property {mongoose.ObjectId}      route           Ride history used route.
 */
const RideHistorySchema = new mongoose.Schema({
  rideScheduleId: { type: mongoose.ObjectId, ref: 'RideSchedule', unique: true, required: true },
  driverId: { type: mongoose.ObjectId, ref: 'User', required: true },
  vehicleId: { type: mongoose.ObjectId, ref: 'Vehicle', required: true },
  passengers: { type: Map, of: PassengerSchema },
  expectedCost: { type: Number, min: 0, required: true },
  realCost: { type: Number, min: 0 },
  startDate: { type: Date, required: true, index: true, default: Date.now },
  endDate: { type: Date },
  route: { type: mongoose.ObjectId, ref: 'RideRoute', required: true },
  isCanceled: { type: Boolean, required: true, default: false }
  // TODO Add statistics
}, {
  timestamps: true
})

RideHistorySchema.index({ driverId: 1, startDate: -1 })

/**
 * The data-layer for a ride history.
 *
 * @example <caption>Example usage of RideHistory model.</caption>
 * let rideHistory1 = new RideHistory({
 *   rideScheduleId: mongoose.Types.ObjectId('5f574fe68b17b9087e60ac0c'),
 *   vehicleId: mongoose.Types.ObjectId('5f58f5876da0db3ca0955318'),
 *   driverId: mongoose.Types.ObjectId('5f58f4ea9207efc18929c52a'),
 *   passengers: {
 *     '5f5b4d9c6b20f6440bc4fd5c': {
 *       userId: mongoose.Types.ObjectId('5f5b4d9c6b20f6440bc4fd5c'),
 *       isPaid: true,
 *       rating: {
 *         rating: 4.5,
 *         comment: 'Loved the ride.'
 *       }
 *     },
 *   },
 *   expectedCost: 15.5,
 *   realCost: 13.2,
 *   endDate: new Date(),
 *   route: mongoose.Types.ObjectId('5f5b4d6f92bef1ff5fa1ed20')
 * });
 */
export default mongoose.model('RideHistory', RideHistorySchema)
