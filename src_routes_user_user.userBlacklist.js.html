src/routes/user/user.userBlacklist.js

/**
 * @module UserUserBlacklistRoute
 * @category Routes
 */

'use strict'

/*
 * ==================================================
 * | Imports                                        |
 * ==================================================
 */
import express from 'express'
import passport from 'passport'

// Controllers
import { UserUserBlacklistController } from '../../controllers/index.js'

/*
 * ==================================================
 * | Constants                                      |
 * ==================================================
 */
const router = express.Router({
  mergeParams: true
})

/*
 * ==================================================
 * | Routes                                         |
 * ==================================================
 */

/**
 * Get specified user's blacklisted users.
 *
 * @name User Blacklist Get All
 * @path {GET} /user/:user/userBlacklist
 * @auth This route requires a valid user's token. If authentication fails it will return a 401 http status code.
 * @header {string}   Authorization The authentication user's token.
 * @params {string}   :user    The id of the user to retrieve the blacklisted users. Use 'me' to reference logged-in user.
 * @code {200} The blacklisted users retrieval succeeded.
 * @code {401} The user authentication has failed.
 * @code {403} The authenticated user does not have permission.
 * @code {404} The requested user id does not exist.
 * @code {500} An internal server error.
 * @response {string}   id                  The blocked user's id.
 * @response {string}   name                The blocked user's name.
 * @response {string}   [avatar.fileName]   The blocked user's avatar file name on the server.
 * @response {string}   [avatar.mimeType]   The blocked user's avatar file mime type.
 * @response {number}   dateOfBlock         The date of the block in timestamp.
 * @example <caption>Possible output:</caption>
 * [
 *   {
 *     "id": "5f732bace3b253c537ea0ffb",
 *     "name": "Henoch Vitureira",
 *     "avatar": {
 *       "fileName": "819cd6318bcaed095ef2ad03ea8159ad",
 *       "mimeType": "image/jpeg"
 *     },
 *     "dateOfBlock": 1601650387
 *   }
 * ]
 */
router.get('/', passport.authenticate('jwt', { session: false }), UserUserBlacklistController.getBlacklistedUsers)

/**
 * Add a user to the specified user's blacklisted users.
 *
 * @name User Blacklist Add Id
 * @path {POST} /user/:user/userBlacklist
 * @auth This route requires a valid user's token. If authentication fails it will return a 401 http status code.
 * @header {string}   Authorization   The authentication user's token.
 * @params {string}   :user   The id of the user to update the blacklisted users. Use 'me' to reference logged-in user.
 * @body {string}   userId   The user id to add to the user blacklist.
 * @code {204} The user addition to the blacklisted users succeeded.
 * @code {401} The user authentication has failed.
 * @code {403} The authenticated user does not have permission.
 * @code {404} The requested user id does not exist.
 * @code {409} The specified user id is already blocked
 * @code {500} An internal server error.
 * @example <caption>Possible input:</caption>
 * {
 *   "userId": "5f732bace3b253c537ea0ffb"
 * }
 */
router.post('/', passport.authenticate('jwt', { session: false }), UserUserBlacklistController.validateAddUserToBlacklist(), UserUserBlacklistController.addUserToBlacklist)

/**
 * Remove a user to the specified user's blacklisted users.
 *
 * @name User Blacklist Remove Id
 * @path {DELETE} /user/:user/userBlacklist
 * @auth This route requires a valid user's token. If authentication fails it will return a 401 http status code.
 * @header {string}   Authorization   The authentication user's token.
 * @params {string}   :user   The id of the user to update the blacklisted users. Use 'me' to reference logged-in user.
 * @body {string}   userId   The user id to remove from the user blacklist.
 * @code {204} The user removal from the blacklisted users succeeded.
 * @code {401} The user authentication has failed.
 * @code {403} The authenticated user does not have permission.
 * @code {404} The requested user id does not exist.
 * @code {409} The specified user id is not blocked
 * @code {500} An internal server error.
 * @example <caption>Possible input:</caption>
 * {
 *   "userId": "5f732bace3b253c537ea0ffb"
 * }
 */
router.delete('/', passport.authenticate('jwt', { session: false }), UserUserBlacklistController.validateRemoveUserFromBlacklist(), UserUserBlacklistController.removeUserFromBlacklist)

/**
 * The router-layer of the user's user blacklist.
 */
export default router
