/**
 * @module UserModel
 * @category Models
 */

'use strict'

/*
 * ==================================================
 * | Imports                                        |
 * ==================================================
 */
import mongoose from 'mongoose'
import mongooseLeanVirtuals from 'mongoose-lean-virtuals'

import Validator from '../utils/validator.js'

// Models
import { UserModel } from './index.js'

// Schemas
import FileSchema from './schemas/file.js'

/*
 * ==================================================
 * | Schemas                                        |
 * ==================================================
 */

/**
 * Blocked user nested schema.
 *
 * @type {mongoose.Schema}
 * @property {mongoose.ObjectId}  userId  Blocked user ObjectId.
 */
const BlockedUserSchema = new mongoose.Schema({
  userId: {
    type: mongoose.ObjectId,
    ref: 'User',
    validate: {
      validator: async function (value) {
        // Validate if the blocking user exists
        const userExists = await UserModel.exists({
          _id: value
        })

        return userExists
      },
      message: 'The specified user id does not exist' // TODO Use i18n
    },
    required: true
  }
}, {
  timestamps: true
})

/**
 * User schema.
 *
 * @type {mongoose.Schema}
 * @property {string}                   aadId             User azure active directory unique id.
 * @property {string}                   givenName         User given name.
 * @property {string}                   [surname]         User surname.
 * @property {string}                   [title]           User job title.
 * @property {string}                   email             User email.
 * @property {string}                   [avatar]          User avatar.
 * @property {string}                   [locale]          User locale.
 * @property {string}                   [fcmToken]        User Firebase Cloud Messaging token.
 * @property {mongoose.ObjectId[]}      [vehicles]        User vehicles.
 * @property {number}                   [level]           User level.
 * @property {number}                   [experience]      User experience.
 * @property {Map<BlockedUserSchema>}   [blockedUsers]    User's blocked users.
 * @property {boolean}                  [isDisabled]      User status (true if deleted).
 */
const UserSchema = new mongoose.Schema({
  aadId: { type: String, unique: true, trim: true, required: true },
  givenName: { type: String, minlength: 3, maxlength: 64, trim: true, required: true },
  surname: { type: String, minlength: 3, maxlength: 64, trim: true },
  title: { type: String, minlength: 3, maxlength: 64, trim: true },
  email: {
    type: String,
    unique: true,
    lowercase: true,
    trim: true,
    validate: {
      validator: Validator.isEmail,
      message: '"{VALUE}" is not a valid email address' // TODO Use i18n
    },
    required: true
  },
  avatar: { type: FileSchema },
  locale: {
    type: String,
    lowercase: true,
    trim: true,
    validate: {
      validator: Validator.isLocale,
      message: '"{VALUE}" is not a valid locale' // TODO Use i18n
    }
  },
  fcmToken: { type: String, unique: true, minlength: 1, trim: true },
  vehicles: { type: [{ type: mongoose.ObjectId, ref: 'Vehicle' }] },
  level: {
    type: Number,
    min: 0,
    validate: {
      validator: Number.isInteger,
      message: '"{VALUE}" is not an integer value' // TODO Use i18n
    },
    required: true,
    default: 0
  },
  experience: {
    type: Number,
    alias: 'xp',
    min: 0,
    validate: {
      validator: Number.isInteger,
      message: '"{VALUE}" is not an integer value' // TODO Use i18n
    },
    required: true,
    default: 0
  },
  blockedUsers: { type: Map, of: BlockedUserSchema },
  isSuspended: { type: Boolean, select: false, required: true, default: false },
  isDisabled: { type: Boolean, select: false, required: true, default: false }
}, {
  timestamps: true
})

UserSchema.index({ givenName: 'text', surname: 'text', title: 'text', email: 'text' })

UserSchema.virtual('name').get(function () {
  return this.givenName + ' ' + this.surname
})

// Set schema plugins
UserSchema.plugin(mongooseLeanVirtuals)

/**
 * The data-layer for a user.
 *
 * @example <caption>Example usage of User model.</caption>
 * let user1 = new User({
 *   aadId: '5f5a506c305ff6bd4475h3s6',
 *   name: 'User1',
 *   surname: 'Surname1',
 *   title: 'Student',
 *   email: 'user1@email.com',
 *   avatar: {
 *     fileName: '2593ee56cc8f8bb16f13995e4f18b746',
 *     originalName: 'WIN_20200729_09_52_41_Pro.jpg',
 *     mimeType: 'image/jpeg',
 *     size: 121976
 *   },
 *   vehicles: [
 *     mongoose.Types.ObjectId('5f5a506c305ff6bd4475b1d1'),
 *     mongoose.Types.ObjectId('5f5a506fc35e608a4832f569')
 *   ],
 *   level: 4,
 *   experience: 2342,
 *   blockedUsers: {
 *     '5f5b444fb804793cf3faed87': {
 *       userId: mongoose.Types.ObjectId('5f5b444fb804793cf3faed87')
 *     }
 *   }
 * });
 */
export default mongoose.model('User', UserSchema)
