/**
 * @module UserVehicleRoute
 * @category Routes
 */

'use strict'

/*
 * ==================================================
 * | Imports                                        |
 * ==================================================
 */
import express from 'express'
import passport from 'passport'

// Controllers
import { UserVehicleController } from '../../controllers/index.js'

/*
 * ==================================================
 * | Constants                                      |
 * ==================================================
 */
const router = express.Router({
  mergeParams: true
})

/*
 * ==================================================
 * | Routes                                         |
 * ==================================================
 */

/**
 * Edit the specified ride vehicle.
 *
 * @name Ride Vehicle Edit
 * @path {PUT} /user/:user/vehicle/:vehicleId
 * @auth This route requires a valid user's token. If authentication fails it will return a 401 http status code.
 * @header {string}   Authorization   The authentication user's token.
 * @params {string}   :user       Use 'me' to reference the logged-in user or specify a user's id to act on the behalf of that user.
 * @params {string}   :vehicleId    The ride vehicle's id of the editing ride vehicle.
 * @body {string}   alias   The new alias of the vehicle being edited.
 * @code {204} The vehicle edition succeeded.
 * @code {401} The user authentication has failed.
 * @code {403} The authenticated user does not have permission.
 * @code {404} The requested vehicle id for the specified user id does not exist.
 * @code {409} The user does not has a vehicle with the provided Id or the user already has a vehicle with the same alias.
 * @code {422} The request content is unable to be processed due to invalid fields.
 * @code {500} An internal server error.
 * @example <caption>Possible input:</caption>
 * {
 *   "alias": "New vehicle ALIAS"
 * }
 */
router.put('/:vehicleId', passport.authenticate('jwt', { session: false }), UserVehicleController.validateEditVehicle(), UserVehicleController.editVehicle)

/**
 * Delete the specified ride vehicle.
 *
 * @name Ride Vehicle Delete
 * @path {DELETE} /user/:user/vehicle/:vehicleId
 * @auth This route requires a valid user's token. If authentication fails it will return a 401 http status code.
 * @header {string}   Authorization   The authentication user's token.
 * @params {string}   :user       Use 'me' to reference the logged-in user or specify a user's id to act on the behalf of that user.
 * @params {string}   :vehicleId    vehicle id of the deleting vehicle.
 * @code {204} The vehicle deletion succeeded.
 * @code {401} The user authentication has failed.
 * @code {403} The authenticated user does not have permission.
 * @code {404} The requested vehicle id for the specified user id does not exist.
 * @code {422} The request content is unable to be processed due to invalid fields.
 * @code {500} An internal server error.
 */
router.delete('/:vehicleId', passport.authenticate('jwt', { session: false }), UserVehicleController.validateDeleteVehicle(), UserVehicleController.deleteVehicle)

/**
 * Renew the specified ride vehicle.
 *
 * @name Ride Vehicle Create
 * @path {POST} /user/:user/vehicle
 * @auth This route requires a valid user's token. If authentication fails it will return a 401 http status code.
 * @header {string}   Authorization   The authentication user's token.
 * @params {string}   :user       Use 'me' to reference the logged-in user or specify a user's id to act on the behalf of that user.
 * @code {201} The vehicle was created.
 * @code {401} The user authentication has failed.
 * @code {409} A vehicle with the same alias already exists for the authenticated user.
 * @code {403} The authenticated user does not have permission.
 * @code {422} The request content is unable to be processed due to invalid fields.
 * @code {500} An internal server error.
 * @example <caption>Possible input of a vehicle:</caption>
 *   {
 *       "alias": "SOME ALIAS",
 *       "fuelType": 1,
 *       "consumption": 12,
 *       "emissions": 24,
 *       "seats": 5,
 *   }
 */
router.post('/', passport.authenticate('jwt', { session: false }), UserVehicleController.validateCreateVehicle(), UserVehicleController.createVehicle)

/**
 * Get specified vehicle details.
 *
 * @name Ride Vehicle Get Id
 * @path {GET} /user/:user/vehicle/:vehicleId
 * @auth This route requires a valid user's token. If authentication fails it will return a 401 http status code.
 * @header {string}   Authorization   The authentication user's token.
 * @params {string}   :user       Use 'me' to reference the logged-in user or specify a user's id to act on the behalf of that user.
 * @params {string}   :vehicleId    The retrieving vehicle's id.
 * @code {200} The user's vehicle vehicle retrieval succeeded.
 * @code {401} The user authentication has failed.
 * @code {403} The authenticated user does not have permission.
 * @code {404} The requested vehicle id does not exist.
 * @code {422} The request content is unable to be processed due to invalid fields.
 * @code {500} An internal server error.
 * @response {string}     vehicle.id                                  The ride vehicle vehicle's id.
 * @response {number}     vehicle.fuelType                            The ride vehicle vehicle's fuel type. Possible values: 1 (Gasoline), 2 (Diesel), 3 (Gpl), 4 (Electric).
 * @response {number}     vehicle.consumption                         The ride vehicle vehicle's consumption in l/100 km or kWh/100 km..
 * @response {number}     vehicle.emissions                           The ride vehicle vehicle's CO2 emissions in g/km.
 * @response {number}     vehicle.seats                               The ride vehicle vehicle's seats.
 * @response {number}     vehicle.remainingSeats                      The ride vehicle vehicle's remaining seats.
 * @example <caption>Possible output:</caption>
 *    {
 *       "id": "5f7c7476ba695503f0845454",
 *       "alias": "SOME ALIAS"
 *       "fuelType": 1,
 *       "consumption": 12,
 *       "emissions": 24,
 *       "seats": 5,
 *       "remainingSeats": 4
 *     }
 */
router.get('/:vehicleId', passport.authenticate('jwt', { session: false }), UserVehicleController.validateGetVehicleDetail(), UserVehicleController.getVehicleDetail)

/**
 * Get all vehicles of a user.
 *
 * @name Ride Vehicle Get All
 * @path {GET} /user/:user/vehicle
 * @auth This route requires a valid user's token. If authentication fails it will return a 401 http status code.
 * @header {string}   Authorization   The authentication user's token.
 * @params {string}   :user       Use 'me' to reference the logged-in user or specify a user's id to act on the behalf of that user.
 * @code {200} The user's vehicles retrieval succeeded.
 * @code {401} The user authentication has failed.
 * @code {403} The authenticated user does not have permission.
 * @code {404} The requested vehicle id does not exist.
 * @code {422} The request content is unable to be processed due to invalid fields.
 * @code {500} An internal server error.
 * @response {string}     vehicle.id                                  The ride vehicle vehicle's id.
 * @response {number}     vehicle.fuelType                            The ride vehicle vehicle's fuel type. Possible values: 1 (Gasoline), 2 (Diesel), 3 (Gpl), 4 (Electric).
 * @response {number}     vehicle.consumption                         The ride vehicle vehicle's consumption in l/100 km or kWh/100 km..
 * @response {number}     vehicle.emissions                           The ride vehicle vehicle's CO2 emissions in g/km.
 * @response {number}     vehicle.seats                               The ride vehicle vehicle's seats.
 * @response {number}     vehicle.remainingSeats                      The ride vehicle vehicle's remaining seats.
 * @example <caption>Possible output:</caption>
 *  [
 *     {
 *       "id": "5f7c7476ba695503f0845454",
 *       "alias": "SOME ALIAS"
 *       "fuelType": 1,
 *       "consumption": 12,
 *       "emissions": 24,
 *       "seats": 5,
 *       "remainingSeats": 4
 *     }
 *  ]
 */
router.get('/', passport.authenticate('jwt', { session: false }), UserVehicleController.getVehicles)

/**
 * The router-layer of the user's ride vehicle.
 */
export default router
